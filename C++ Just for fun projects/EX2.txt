	int ejm;

	cout << "eem: ";

	cout << "jma: ";
	cin >> ejm;

	string e = string(ee, ' ');

	for (int i = ei; i <= eim; ++i) {
		cout << l << endl;
		cout << e << "main " << i << endl;
		if (i <= 9) {
			cout << "lot10" << endl;
			cout << l << endl;
		}
		else if (i == 10) {
			cout << "eq10" << endl;
			cout << l << endl;
		}

		}

			cout << " su1 - " << j << endl;
